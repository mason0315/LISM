## 1. 项目概述

本项目是一个基于Spring Cloud Alibaba生态的图书管理系统，采用微服务架构设计，整合了多种异构数据源（MySQL、Neo4j图数据库），并集成了AI能力（大语言模型、人脸识别）提供智能化服务。

#### 2.1.2 核心设计原则

**1. 单一职责原则 (SRP)**
每个微服务只负责一个明确的业务领域：
- `auth-service`: 用户认证与授权
- `user-service`: 用户信息管理
- `book-service`: 图书信息管理
- `borrow-service`: 借阅记录管理
- `face-service`: 人脸识别服务

**2. 数据库独立原则**
每个微服务拥有独立的数据库实例，避免数据耦合：

```yaml
# book-service: book_db
spring:
  datasource:
    url: jdbc:mysql://localhost:3306/book_db
    
# user-service: user_db
spring:
  datasource:
    url: jdbc:mysql://localhost:3306/user_db
    
# auth-service: auth_db
spring:
  datasource:
    url: jdbc:mysql://localhost:3306/auth_db
```

**3. API网关统一入口**
所有外部请求通过Gateway统一接入，实现：
- 统一认证（JWT验证）
- 路由转发
- 负载均衡
- 限流熔断

#### 3.1.2 JWT认证过滤器实现

```java
@Slf4j
@Component
public class JwtAuthenticationFilter extends 
        AbstractGatewayFilterFactory<JwtAuthenticationFilter.Config> {
    
    // 白名单路径
    private static final List<String> WHITE_LIST = Arrays.asList(
        "/auth/login",
        "/auth/register",
        "/swagger-ui",
        "/v3/api-docs"
    );
    
    @Override
    public GatewayFilter apply(Config config) {
        return (exchange, chain) -> {
            ServerHttpRequest request = exchange.getRequest();
            String path = request.getURI().getPath();
            
            // 白名单放行
            if (isWhiteList(path)) {
                return chain.filter(exchange);
            }
            
            // 提取并验证Token
            String authHeader = request.getHeaders()
                    .getFirst(HttpHeaders.AUTHORIZATION);
            
            if (authHeader == null || !authHeader.startsWith("Bearer ")) {
                return unauthorized(exchange, "缺少认证令牌");
            }
            
            String token = authHeader.substring(7);
            
            try {
                Claims claims = JwtUtil.parseToken(token);
                
                // 将用户信息传递到下游服务
                ServerHttpRequest mutatedRequest = request.mutate()
                    .header("X-User-Id", String.valueOf(claims.get("userId")))
                    .header("X-User-Name", claims.get("username", String.class))
                    .header("X-User-Role", String.valueOf(claims.get("role")))
                    .build();
                
                return chain.filter(exchange.mutate()
                        .request(mutatedRequest).build());
                
            } catch (ExpiredJwtException e) {
                return unauthorized(exchange, "认证令牌已过期");
            }
        };
    }
}
```

@Configuration
public class FeignConfig {

    public RequestInterceptor feignRequestInterceptor() {
        return template -> {
            // 传递JWT Token
            ServletRequestAttributes attributes = 
                (ServletRequestAttributes) RequestContextHolder
                    .getRequestAttributes();
            
            if (attributes != null) {
                HttpServletRequest request = attributes.getRequest();
                String token = request.getHeader("Authorization");
                if (token != null) {
                    template.header("Authorization", token);
                }
            }

#### 4.2.2 智能查询引擎

```python
class Neo4jHandler:
    """Neo4j知识图谱处理器"""
    
    def extract_keywords(self, question: str) -> List[str]:
        """
        从问题中提取关键词
        使用正则表达式和停用词过滤
        """
        stop_words = {'的', '了', '和', '是', '在', '有', ...}
        
        # 提取书名（形如《书名》）
        book_titles = re.findall(r'《([^》]+)》', question)
        
        # 提取查询词
        query_terms = ['作者', '年代', '主题', '人物', '背景']
        keywords = [term for term in query_terms if term in question]
        
        return keywords + book_titles
    
    def query_expanded_context(self, keywords: List[str], depth: int = 2):
        """
        扩展查询：获取关键词相关的多跳关系
        """
        query = """
        MATCH (n)
        WHERE ANY(prop IN keys(n) WHERE toString(n[prop]) CONTAINS $keyword)
        OPTIONAL MATCH (n)-[r*1..%d]-(related)
        RETURN n, r, related
        LIMIT 50
        """ % depth
        
        with self.driver.session() as session:
            result = session.run(query, keyword=keywords[0])
            return self.process_result(result)
```

public class FaceUtil {

     * 检测人脸
     */
    public static boolean detectFace(byte[] imageBytes) {
        try {
            Mat image = Imgcodecs.imdecode(
                new MatOfByte(imageBytes), 
                Imgcodecs.IMREAD_COLOR
            );
            
            // 灰度转换
            Mat grayImage = new Mat();
            Imgproc.cvtColor(image, grayImage, Imgproc.COLOR_BGR2GRAY);
            Imgproc.equalizeHist(grayImage, grayImage);
            
            // 人脸检测
            MatOfRect faces = new MatOfRect();
            faceDetector.detectMultiScale(
                grayImage, faces, 
                1.1, 3, 0, 
                new Size(30, 30), 
                new Size()
            );
            
            return faces.toArray().length > 0;
            
        } catch (Exception e) {
            log.error("人脸检测失败", e);
            return false;
        }
    }
    
    /**
     * 提取人脸特征
     */
    public static String extractFeature(byte[] imageBytes) {
        Mat image = Imgcodecs.imdecode(
            new MatOfByte(imageBytes), 
            Imgcodecs.IMREAD_GRAYSCALE
        );
        
        // 调整大小为固定尺寸
        Mat resized = new Mat();
        Imgproc.resize(image, resized, new Size(64, 64));
        
        // 计算特征（简化实现）
        MatOfFloat features = new MatOfFloat();
        // ... 特征提取逻辑
        
        return Base64.getEncoder().encodeToString(features.toArray());
    }
}

def content_based_recommend(user_id: int, top_n: int = 10):
    """
    基于图书内容的推荐
    """
    # 1. 获取用户历史借阅
    user_history = get_user_borrow_history(user_id)
    
    # 2. 提取用户偏好特征
    user_profile = {
        'categories': Counter(book.category for book in user_history),
        'authors': Counter(book.author for book in user_history),
        'keywords': extract_keywords_from_books(user_history)
    }
    
    # 3. 计算图书相似度
    all_books = get_all_books()
    similarities = []
    
    for book in all_books:
        if book not in user_history:
            sim = calculate_similarity(user_profile, book)
            similarities.append((book, sim))
    
    # 4. 返回Top N
    similarities.sort(key=lambda x: x[1], reverse=True)
    return [book for book, _ in similarities[:top_n]]

def collaborative_filtering_recommend(user_id: int, top_n: int = 10):
    """
    基于用户的协同过滤推荐
    """
    # 1. 构建用户-图书评分矩阵
    ratings_matrix = build_ratings_matrix()
    
    # 2. 计算用户相似度
    user_similarities = {}
    target_user = ratings_matrix[user_id]
    
    for other_id, other_vector in ratings_matrix.items():
        if other_id != user_id:
            sim = cosine_similarity(target_user, other_vector)
            user_similarities[other_id] = sim
    
    # 3. 找到最相似的K个用户
    k_nearest = sorted(
        user_similarities.items(), 
        key=lambda x: x[1], 
        reverse=True
    )[:20]
    
    # 4. 生成推荐
    recommendations = {}
    for similar_id, similarity in k_nearest:
        for book_id, rating in ratings_matrix[similar_id].items():
            if book_id not in target_user:
                if book_id not in recommendations:
                    recommendations[book_id] = 0
                recommendations[book_id] += similarity * rating
    
    # 5. 返回Top N
    sorted_recs = sorted(
        recommendations.items(), 
        key=lambda x: x[1], 
        reverse=True
    )
    
    return [book_id for book_id, _ in sorted_recs[:top_n]]

@Service
public class BookService {
    
    @Autowired
    private BookMapper bookMapper;
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    private static final String BOOK_CACHE_KEY = "book:";
    private static final long CACHE_TTL = 3600; // 1小时
    
    /**
     * 带缓存的图书查询
     */
    public Book getBookByTitle(String title) {
        String cacheKey = BOOK_CACHE_KEY + title;
        
        // 1. 查询缓存
        String cached = redisTemplate.opsForValue().get(cacheKey);
        if (cached != null) {
            return JSON.parseObject(cached, Book.class);
        }
        
        // 2. 查询数据库
        Book book = bookMapper.selectById(title);
        
        // 3. 写入缓存
        if (book != null) {
            redisTemplate.opsForValue().set(
                cacheKey, 
                JSON.toJSONString(book),
                CACHE_TTL,
                TimeUnit.SECONDS
            );
        }
        
        return book;
    }
    
    /**
     * 缓存更新
     */
    public void updateBook(Book book) {
        // 1. 更新数据库
        bookMapper.updateById(book);
        
        // 2. 删除缓存（非更新，避免并发问题）
        String cacheKey = BOOK_CACHE_KEY + book.getTitle();
        redisTemplate.delete(cacheKey);
    }
}
